def bubble_sort(arr):
    n = len(arr)
    # 外层循环初始化 i = 0
    i = 0
    while i < n - 1:
        # 内层循环初始化 j = 0
        j = 0
        while j < n - i - 1:
            if arr[j] > arr[j + 1]:
                # 交换 arr[j] 和 arr[j+1]
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
            j += 1
        i += 1
    return arr

# 示例用法
